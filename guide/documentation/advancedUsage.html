# Advanced Usage

We have made further encapsulation based on the basic usage mentioned in the previous chapter, and now we have implemented the scene of pagination, loadMore. If you have good suggestions, welcome to githubopen in new window to submit ISSUE or PR.

# Pagination New in beta.8

We will automatically help you manage the paging information, such as current page number, number of entries per page, the total number of entries, etc. You only need to use the changeCurrent() and changePageSize() provided by us. Easily manage paged data.

Note: We will inject the parameters required for paging in the first position of the params array.

See more available configurations: Pagination API

# Usage

import { usePagination } from 'vue-request';

# Example

You can also use it with ant-design-vue. Here we provide a demo, click to viewopen in new window

View Code
import { defineComponent } from 'vue';
import { usePagination } from 'vue-request';
import Mock from 'mockjs';
import { Loading, Pagination } from 'vant';
import 'vant/lib/pagination/style';
import 'vant/lib/loading/style';

const getData = (): string[] => Mock.mock({ 'array|5': ['@name'] }).array;
function testService() {
  return new Promise<{ total: number; result: string[] }>(resolve => {
    setTimeout(() => {
      resolve({
        total: 100,
        result: getData(),
      });
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { data, current, total, loading, pageSize, changeCurrent } = usePagination(testService);

    return () => (
      <div>
        <div style="height: 140px">
          {loading.value ? (
            <Loading />
          ) : (
            <ul>
              {data.value?.result.map(i => (
                <li key={i}>{i}</li>
              ))}
            </ul>
          )}
        </div>
        <Pagination
          modelValue={current.value}
          total-items={total.value}
          show-page-size={pageSize.value}
          force-ellipses
          onChange={(page: number) => changeCurrent(page)}
        />
      </div>
    );
  },
});

# LoadMore New in beta.11

Note

In LoadMore mode, only Request Method is supported.

We will automatically help you manage the list data. The data structure returned by Service must contain {list: LR[]}. If not, you can specify another key name with listKey or convert it once with formatResult(). Then we can get the requested data list collection through the returned dataList variable, and then use loadMore() to trigger loading more.

In addition, we will pass in the first parameter of the Request Method {data: R, dataList: LR}. You can decide the parameters of the next request based on these parameters.

See more available configurations: LoadMore API

# Usage

import { useLoadMore } from 'vue-request';

# Example

You can also use it with ant-design-vue. Here we provide a demo, click to viewopen in new window

View Code
import { defineComponent, ref } from 'vue';
import { useLoadMore } from 'vue-request';
import Mock from 'mockjs';
import { Cell, List, Button, PullRefresh } from 'vant';
import 'vant/es/list/style';
import 'vant/es/cell/style';
import 'vant/es/button/style';
import 'vant/es/pull-refresh/style';

function generateData(current: number, total = 10) {
  let list: string[] = [];
  if (current <= total) {
    list = Mock.mock({
      'list|10': ['@name'],
    }).list;
  } else {
    list = [];
  }
  return {
    current,
    total,
    list,
  };
}
type APIReturnType = ReturnType<typeof generateData>;
type rawDataType = {
  data: APIReturnType;
  dataList: APIReturnType['list'];
};

function testService(rawData: rawDataType) {
  const current = (rawData?.data?.current || 0) + 1;
  return new Promise<APIReturnType>(resolve => {
    setTimeout(() => {
      resolve(generateData(current));
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { loadMore, loadingMore, dataList, noMore, refreshing, reload, refresh } = useLoadMore<
      APIReturnType,
      any,
      APIReturnType['list']
    >(testService, {
      isNoMore: d => d?.current >= d?.total,
    });

    const onRefresh = () => {
      document.querySelector('.list')?.scrollTo({ top: 0, behavior: 'smooth' });
      refresh();
    };

    return () => (
      <div>
        <Button
          style="margin-bottom: 20px"
          type="warning"
          size="mini"
          onClick={() => {
            reload();
          }}
        >
          Reload
        </Button>
        <Button style="margin-bottom: 20px" type="warning" size="mini" onClick={onRefresh}>
          Refresh
        </Button>
        <div class="list" style="height: 300px; overflow-y: scroll">
          <PullRefresh modelValue={refreshing.value} onRefresh={() => refresh()}>
            <List
              loading={loadingMore.value}
              loading-text="loading..."
              finished={noMore.value}
              finished-text="no more"
              onLoad={() => {
                loadMore();
              }}
            >
              {dataList.value?.map((i, idx) => (
                <Cell key={i} value={`${idx}  ${i}`} />
              ))}
            </List>
          </PullRefresh>
        </div>
      </div>
    );
  },
});
